--help     -  к любой команде можно дописать --help выйдет описание как ее использовать
cd test    -  переход в папку (если это папка и если она существует)
cd ..      -  переход в (предыдущую папку)
cd -       -  переход назад
cd .       -  переход в текущую папку
cd         -  переход в домашнюю папку (~)
pwd        -  показывает абсолютный путь до текущей папки
mkdir test -  создание новой папки
mkdir test/new_dir  - создание новой папки new_dir в уже существующей папки
touch  test.txt     - создание файла test.txt
rm test.txt         - удаление файла
rm -rf              - удаление папки совсем его содержимым (не в корзину а навсегда)
sudo rm -rf         - никогда не использовать (удаление операционки)
cat linux.txt       - просмотр содержимого файла
ls test             - просмотр всех файлов в папке внутри папки test
ls -a test          - просмотр всех файлов включая скрытых
ls -la              - просмотр
ls -l               - просмотр всех файлов с подробной информацией
nano text.txt       - открыть файл в редакторе(nano)
cntrl S           - чтоб сохранить
cntrl X           - чтоб выйти из редактора
